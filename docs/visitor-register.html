document.getElementById('registrationForm').addEventListener('submit', async function(e) { e.preventDefault(); // Show loading document.getElementById('loading').classList.remove('hidden'); const formData = new FormData(this); const data = Object.fromEntries(formData); try { const response = await fetch('https://n8n.pintarweb.com/webhook/visitor-register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); console.log('Response status:', response.status); const result = await response.json(); console.log('Full response:', result); if (response.ok) { // DEBUG: Check what we're working with console.log('Name from form:', data.name); console.log('Name from response:', result.name); // Always use the name from the FORM (most reliable) const visitorName = data.name || result.name || 'Visitor'; const passUrl = result.pass_url || `https://pintarweb.github.io/event-data-system/pass.html?visitor_id=${result.visitor_id}`; console.log('Final redirect to:', `${passUrl}&name=${encodeURIComponent(visitorName)}`); window.location.href = `${passUrl}&name=${encodeURIComponent(visitorName)}`; } else { alert('Registration failed. Please try again.'); } } catch (error) { console.error('Error:', error); alert('Network error. Please check your connection.'); } finally { document.getElementById('loading').classList.add('hidden'); } });
